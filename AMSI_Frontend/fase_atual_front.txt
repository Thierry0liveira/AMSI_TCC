--------------------------------------------------------------------------------
  ROTAS (App.jsx)
--------------------------------------------------------------------------------

  /                              → LoginPage             (pública)
  /trocar-senha                  → TrocarSenhaPage       (privada, aceita ?senha=)
  /home                          → HomePage              (admin)
  /dashboard                     → Dashboard             (admin)
  /tipo_lancamento               → ListaLancamentosPage  (admin)
  /cliente_fornecedor            → ClientListPage        (admin)
  /cliente_fornecedor/novo       → ClientRegisterPage    (admin)
  /cliente_fornecedor/:id/editar → ClientEditPage        (admin)
  /usuarios                      → UserListPage          (admin)
  /cadastro                      → UserRegisterPage      (admin — mantida)
